## 1) Basic Pattern

When a scene needs post-processing (blur, opacity, color correction, etc.), the standard flow is:

```
Scene -> Tex -> Pass -> Present
```

1. Render the scene to a texture using `from="scene:scene_id"`
2. Apply one or more compute passes
3. Present the final output texture

## 2) Scene with Opacity Pass

```xml
<Graph fps={30} duration="3s" size={[1920,1080]}>
  <Background color="#000000" />

  <Scene id="scene0">
    <Timeline>
      <Track id="main" space="world" z="0">
        <Sequence from="0s" duration="3s" out="hold">
          <Layer>
            <Text value="HELLO WORLD" x="center" y="center" fontSize="96" color="#ffffff" />
          </Layer>
        </Sequence>
      </Track>
    </Timeline>
  </Scene>

  <Tex id="src" fmt="rgba16f" from="scene:scene0" />
  <Tex id="out" fmt="rgba16f" size={[1920,1080]} />
  <Pass id="fx_opacity" kind="compute" effect="opacity" in={["src"]} out={["out"]} params={{ opacity: "0.8" }} />
  <Present from="out" />
</Graph>
```

## 4) Scene + World + Composite

Combine scene and world graphs into one composition:

```xml
<Graph fps={30} duration="4s" size={[1280,720]} renderSize={[1280,720]}>
  <Background color="#FFFFFF" />

  <Scene id="title_scene">
    <Timeline>
      <Track id="main" space="world" z="0">
        <Sequence from="0s" duration="4s" out="hold">
          <Layer>
            <Text value="Title" x="center" y="center" fontSize="48" color="#111827" />
          </Layer>
        </Sequence>
      </Track>
    </Timeline>
  </Scene>

  <World id="world_actor">
    <Camera yaw="0" pitch="0" zoom="1" />
    <Actor id="actor" model="character.glb" x="0" y="0" z="0" scale="0.35" />
  </World>

  <Process id="final_composite">
    <Tex id="scene_tex" fmt="rgba16f" from="scene:title_scene" />
    <Tex id="world_tex" fmt="rgba16f" from="world:world_actor" />
    <Tex id="merged" fmt="rgba16f" size={[1280,720]} />
    <Pass id="world_over_scene" kind="compute" effect="over" in={["scene_tex", "world_tex"]} out={["merged"]} />
  </Process>

  <Present from="final_composite" />
</Graph>
```

## 5) Multi-pass Effect Chain

Chain multiple effects on a scene:

```xml
<Graph fps={30} duration="3s" size={[1920,1080]}>
  <Background color="#000000" />

  <Scene id="scene0">
    <Timeline>
      <Track id="main" space="world" z="0">
        <Sequence from="0s" duration="3s" out="hold">
          <Layer>
            <Text value="Effects" x="center" y="center" fontSize="96" color="#ffffff" />
          </Layer>
        </Sequence>
      </Track>
    </Timeline>
  </Scene>

  <Tex id="scene_src" fmt="rgba16f" from="scene:scene0" />
  <Tex id="blur" fmt="rgba16f" size={[1920,1080]} />
  <Tex id="color" fmt="rgba16f" size={[1920,1080]} />
  <Tex id="out" fmt="rgba16f" size={[1920,1080]} />

  <!-- Pass 1: blur -->
  <Pass id="fx_blur" kind="compute" effect="gaussian_5tap_h" in={["scene_src"]} out={["blur"]} params={{ sigma: "6.0" }} />

  <!-- Pass 2: color adjustment -->
  <Pass id="fx_color" kind="compute" effect="color_core" in={["blur"]} out={["color"]} params={{ contrast: "1.2", brightness: "0.05" }} />

  <!-- Pass 3: final opacity -->
  <Pass id="fx_opacity" kind="compute" effect="opacity" in={["color"]} out={["out"]} params={{ opacity: "0.9" }} />

  <Present from="out" />
</Graph>
```

## 6) Scene with Precompose

Use `<Precompose>` to inline a sub-scene:

```xml
<Graph fps={30} duration="3s" size={[1920,1080]}>
  <Scene id="main_scene">
    <Timeline>
      <Track id="main" space="world" z="0">
        <Sequence from="0s" duration="3s" out="hold">
          <Layer>
            <Precompose from="scene:sub_scene" x="0" y="0" width="960" height="1080" />
            <Text value="Main Title" x="1200" y="center" fontSize="48" color="#ffffff" />
          </Layer>
        </Sequence>
      </Track>
    </Timeline>
  </Scene>

  <Scene id="sub_scene">
    <Timeline>
      <Track id="main" space="world" z="0">
        <Sequence from="0s" duration="3s" out="hold">
          <Layer>
            <Rect x="0" y="0" width="960" height="1080" color="#1a1a2e" />
            <Text value="Sub" x="center" y="center" fontSize="36" color="#ffffff" />
          </Layer>
        </Sequence>
      </Track>
    </Timeline>
  </Scene>

  <Present from="main_scene" />
</Graph>
```

## 7) ACP Tool Workflow For Compositions

1. Call `anica.motionloom/set_scene_script`
   - pass generated DSL string
   - set `apply_now=true`
2. If user asked for output video / render:
   - call `anica.motionloom/render_scene`
   - default `mode="gpu"` on macOS
   - use `mode="gpu_prores"` or `mode="compatibility_cpu"` only when explicitly requested

## ACP Reply (Short)

For scene + post-process: `Scene -> Tex -> Pass -> Present`. Use `from="scene:scene_id"` to route the scene into a texture. Chain multiple passes by routing `out` of one pass to `in` of the next. Use `<Precompose>` to inline sub-scenes. For scene + world, use `Process` with `from="scene:..."` and `from="world:..."` inputs.
